1""" 

2 This module provides one-time keys. 

3 

4 There are two types of security keys: 

5 

6 1. If :meth:create is called without arguments, it returns the current session CSRF token. Repeated calls to 

7 :meth:create will return the same CSRF token (for the same session) until that token has been redeemed. 

8 This security key will be valid as long the session is active, and it's not possible to store data along with 

9 that key. These are usually used as a CSRF token. 

10 This has been changed from ViUR2 - where it was possible to create a arbitrary number of security keys per 

11 session. 

12 2. If :meth:create is called with a duration (and optional kwargs values), it will create a security key 

13 that is *not* bound to the current session, but it's possible to store custom data (provided by kwargs). 

14 As these are not bound to the session, each call to :meth:create will yield a new token. 

15 These are used if it's expected that the token may be redeemed on a different device (e.g. when sending an 

16 email address confirmation) 

17 

18 ..note: 

19 There's also a hidden 3rd type of security-key: The sessions static security key. 

20 

21 This key is only revealed once (during login, as the protected header Sec-X-ViUR-StaticSKey). 

22 

23 This can be used instead of the one-time sessions security key by sending it back as the same protected HTTP 

24 header and setting the skey value to "staticSessionKey". This is only intended for non-web-browser, 

25 programmatic access (admin tools, import tools etc.) where CSRF attacks are not applicable. 

26 

27 Therefor that header is prefixed with "Sec-" - so it cannot be read or set using JavaScript. 

28"""

37def create(duration: Union[None, int] = None, **custom_data) -> str: 

38 """ 

39 Creates a new one-time security key or returns the current sessions CSRF-token. 

40 

41 The custom data (given as **custom_data) that can be stored with the key. 

42 Any data provided must be serializable by the datastore. 

43 

44 :param duration: Make this key valid for a fixed timeframe of seconds (and independent of the current session) 

45 :returns: The new one-time key, which is a randomized string. 

46 """ 

47 if not duration: 

48 if custom_data: 

49 raise ValueError("Providing any custom_data is not allowed when session security key is wanted") 

50 

51 return current.session.get().getSecurityKey() 

52 

53 elif "until" in custom_data: 

54 raise ValueError("The 'until' property is reserved and cannot be used in the custom_data") 

55 

56 key = utils.generateRandomString() 

57 

58 entity = db.Entity(db.Key(SECURITYKEY_KINDNAME, key)) 

59 entity |= custom_data 

60 

61 entity["until"] = utils.utcNow() + timedelta(seconds=int(duration)) 

62 db.Put(entity) 

63 

64 return key

67def validate(key: str, useSessionKey: bool) -> Union[bool, db.Entity]: 

68 """ 

69 Validates a security key. 

70 

71 If useSessionKey is True, the key is expected to be the session's current security key 

72 or its static security key. 

73 

74 Otherwise, it must be a key created with a duration, so that it is session independent. 

75 

76 :param key: The key to be validated 

77 :param useSessionKey: If True, validate against the session's skey, otherwise lookup an unbound key 

78 :returns: False if the key was not valid for whatever reasons, the data (given during createSecurityKey) as 

79 dictionary or True if the dict is empty (or useSessionKey was True). 

80 """ 

81 if useSessionKey: 

82 session = current.session.get() 

83 if key == "staticSessionKey": 

84 skey_header_value = current.request.get().request.headers.get("Sec-X-ViUR-StaticSKey") 

85 if skey_header_value and session.validateStaticSecurityKey(skey_header_value): 

86 return True 

87 

88 elif session.validateSecurityKey(key): 

89 return True 

90 

91 return False 

92 

93 if not key or not (entity := db.Get(db.Key(SECURITYKEY_KINDNAME, key))): 

94 return False 

95 

96 db.Delete(entity) 

97 

98 # Key has expired? 

99 if entity["until"] < utils.utcNow(): 

100 return False 

101 

102 del entity["until"] 

103 

104 return entity or True
