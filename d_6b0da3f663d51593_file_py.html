12def ensureDerived(key: db.Key, srcKey, deriveMap: Dict[str, Any], refreshKey: db.Key = None): 

13 """ 

14 Ensure that pending thumbnails or other derived Files are build 

15 :param key: DB-Key of the file-object on which we should update the derivemap 

16 :param srcKey: Prefix for a (hopefully) stable key to prevent rebuilding derives over and over again 

17 :param deriveMap: List of DeriveDicts we should build/update 

18 :param refreshKey: If set, we'll fetch and refresh the skeleton after building new derives 

19 """ 

20 from viur.core.skeleton import skeletonByKind, updateRelations 

21 deriveFuncMap = conf["viur.file.derivers"] 

22 skel = skeletonByKind("file")() 

23 if not skel.fromDB(key): 

24 logging.info("File-Entry went missing in ensureDerived") 

25 return 

26 if not skel["derived"]: 

27 logging.info("No Derives for this file") 

28 skel["derived"] = {} 

29 skel["derived"]["deriveStatus"] = skel["derived"].get("deriveStatus") or {} 

30 skel["derived"]["files"] = skel["derived"].get("files") or {} 

31 resDict = {} # Will contain new or updated resultDicts that will be merged into our file 

32 for calleeKey, params in deriveMap.items(): 

33 fullSrcKey = "%s_%s" % (srcKey, calleeKey) 

34 paramsHash = sha256(str(params).encode("UTF-8")).hexdigest() # Hash over given params (dict?) 

35 if skel["derived"]["deriveStatus"].get(fullSrcKey) != paramsHash: 

36 if calleeKey not in deriveFuncMap: 

37 logging.warning("File-Deriver %s not found - skipping!" % calleeKey) 

38 continue 

39 callee = deriveFuncMap[calleeKey] 

40 callRes = callee(skel, skel["derived"]["files"], params) 

41 if callRes: 

42 assert isinstance(callRes, list), "Old (non-list) return value from deriveFunc" 

43 resDict[fullSrcKey] = {"version": paramsHash, "files": {}} 

44 for fileName, size, mimetype, customData in callRes: 

45 resDict[fullSrcKey]["files"][fileName] = { 

46 "size": size, 

47 "mimetype": mimetype, 

48 "customData": customData 

49 } 

50 

51 def updateTxn(key, resDict): 

52 obj = db.Get(key) 

53 if not obj: # File-object got deleted during building of our derives 

54 return 

55 obj["derived"] = obj.get("derived") or {} 

56 obj["derived"]["deriveStatus"] = obj["derived"].get("deriveStatus") or {} 

57 obj["derived"]["files"] = obj["derived"].get("files") or {} 

58 for k, v in resDict.items(): 

59 obj["derived"]["deriveStatus"][k] = v["version"] 

60 for fileName, fileDict in v["files"].items(): 

61 obj["derived"]["files"][fileName] = fileDict 

62 db.Put(obj) 

63 

64 if resDict: # Write updated results back and queue updateRelationsTask 

65 db.RunInTransaction(updateTxn, key, resDict) 

66 # Queue that updateRelations call at least 30 seconds into the future, so that other ensureDerived calls from 

67 # the same FileBone have the chance to finish, otherwise that updateRelations Task will call postSavedHandler 

68 # on that FileBone again - re-queueing any ensureDerivedCalls that have not finished yet. 

69 updateRelations(key, time() + 1, "derived", _countdown=30) 

70 if refreshKey: 

71 def refreshTxn(): 

72 skel = skeletonByKind(refreshKey.kind)() 

73 if not skel.fromDB(refreshKey): 

74 return 

75 skel.refresh() 

76 skel.toDB(update_relations=False) 

77 

78 db.RunInTransaction(refreshTxn)

121 def isInvalid(self, value): 

122 if self.validMimeTypes: 

123 mimeType = value["dest"]["mimetype"] 

124 for checkMT in self.validMimeTypes: 

125 checkMT = checkMT.lower()
