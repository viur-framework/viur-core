19def ensureDerived(key: db.Key, srcKey, deriveMap: Dict[str, Any], refreshKey: db.Key = None): 

20 r""" 

21 The function is a deferred function that ensures all pending thumbnails or other derived files 

22 are built. It takes the following parameters: 

23 

24 :param db.key key: The database key of the file-object that needs to have its derivation map 

25 updated. 

26 :param str srcKey: A prefix for a stable key to prevent rebuilding derived files repeatedly. 

27 :param Dict[str,Any] deriveMap: A list of DeriveDicts that need to be built or updated. 

28 :param db.Key refreshKey: If set, the function fetches and refreshes the skeleton after 

29 building new derived files. 

30 

31 The function works by fetching the skeleton of the file-object, checking if it has any derived 

32 files, and updating the derivation map accordingly. It iterates through the deriveMap items and 

33 calls the appropriate deriver function. If the deriver function returns a result, the function 

34 creates a new or updated resultDict and merges it into the file-object's metadata. Finally, 

35 the updated results are written back to the database and the updateRelations function is called 

36 to ensure proper relations are maintained. 

37 """ 

38 from viur.core.skeleton import skeletonByKind, updateRelations 

39 deriveFuncMap = conf["viur.file.derivers"] 

40 skel = skeletonByKind("file")() 

41 if not skel.fromDB(key): 

42 logging.info("File-Entry went missing in ensureDerived") 

43 return 

44 if not skel["derived"]: 

45 logging.info("No Derives for this file") 

46 skel["derived"] = {} 

47 skel["derived"]["deriveStatus"] = skel["derived"].get("deriveStatus") or {} 

48 skel["derived"]["files"] = skel["derived"].get("files") or {} 

49 resDict = {} # Will contain new or updated resultDicts that will be merged into our file 

50 for calleeKey, params in deriveMap.items(): 

51 fullSrcKey = "%s_%s" % (srcKey, calleeKey) 

52 paramsHash = sha256(str(params).encode("UTF-8")).hexdigest() # Hash over given params (dict?) 

53 if skel["derived"]["deriveStatus"].get(fullSrcKey) != paramsHash: 

54 if calleeKey not in deriveFuncMap: 

55 logging.warning("File-Deriver %s not found - skipping!" % calleeKey) 

56 continue 

57 callee = deriveFuncMap[calleeKey] 

58 callRes = callee(skel, skel["derived"]["files"], params) 

59 if callRes: 

60 assert isinstance(callRes, list), "Old (non-list) return value from deriveFunc" 

61 resDict[fullSrcKey] = {"version": paramsHash, "files": {}} 

62 for fileName, size, mimetype, customData in callRes: 

63 resDict[fullSrcKey]["files"][fileName] = { 

64 "size": size, 

65 "mimetype": mimetype, 

66 "customData": customData 

67 } 

68 

69 def updateTxn(key, resDict): 

70 obj = db.Get(key) 

71 if not obj: # File-object got deleted during building of our derives 

72 return 

73 obj["derived"] = obj.get("derived") or {} 

74 obj["derived"]["deriveStatus"] = obj["derived"].get("deriveStatus") or {} 

75 obj["derived"]["files"] = obj["derived"].get("files") or {} 

76 for k, v in resDict.items(): 

77 obj["derived"]["deriveStatus"][k] = v["version"] 

78 for fileName, fileDict in v["files"].items(): 

79 obj["derived"]["files"][fileName] = fileDict 

80 db.Put(obj) 

81 

82 if resDict: # Write updated results back and queue updateRelationsTask 

83 db.RunInTransaction(updateTxn, key, resDict) 

84 # Queue that updateRelations call at least 30 seconds into the future, so that other ensureDerived calls from 

85 # the same FileBone have the chance to finish, otherwise that updateRelations Task will call postSavedHandler 

86 # on that FileBone again - re-queueing any ensureDerivedCalls that have not finished yet. 

87 updateRelations(key, time() + 1, "derived", _countdown=30) 

88 if refreshKey: 

89 def refreshTxn(): 

90 skel = skeletonByKind(refreshKey.kind)() 

91 if not skel.fromDB(refreshKey): 

92 return 

93 skel.refresh() 

94 skel.toDB(update_relations=False) 

95 

96 db.RunInTransaction(refreshTxn)

182 def isInvalid(self, value): 

183 """ 

184 Checks if the provided value is invalid for this bone based on its MIME type and file size. 

185 

186 :param dict value: The value to check for validity. 

187 :returns: None if the value is valid, or an error message if it is invalid. 

188 """ 

189 if self.validMimeTypes: 

190 mimeType = value["dest"]["mimetype"] 

191 for checkMT in self.validMimeTypes: 

192 checkMT = checkMT.lower() 

193 if checkMT == mimeType or checkMT.endswith("*") and mimeType.startswith(checkMT[:-1]): 

194 break 

195 else: 

196 return "Invalid filetype selected" 

197 if self.maxFileSize: 

198 if value["dest"]["size"] > self.maxFileSize: 

199 return "File too large."

254 def refresh(self, skel, boneName): 

255 r""" 

256 Refreshes the FileBone by recreating file entries if needed and importing blobs from ViUR 2. 

257 

258 :param SkeletonInstance skel: The skeleton instance this bone belongs to. 

259 :param str boneName: The name of the bone. 

260 

261 This method defines an inner function, recreateFileEntryIfNeeded(val), which is responsible 

262 for recreating the weak file entry referenced by the relation in val if it doesn't exist 

263 (e.g., if it was deleted by ViUR 2). It initializes a new skeleton for the "file" kind and 

264 checks if the file object already exists. If not, it recreates the file entry with the 

265 appropriate properties and saves it to the database. 

266 

267 The main part of the refresh method calls the superclass's refresh method and checks if the 

268 configuration contains a ViUR 2 import blob source. If it does, it iterates through the file 

269 references in the bone value, imports the blobs from ViUR 2, and recreates the file entries if 

270 needed using the inner function. 

271 """ 

272 from viur.core.skeleton import skeletonByKind 

273 

274 def recreateFileEntryIfNeeded(val): 

275 # Recreate the (weak) filenetry referenced by the relation *val*. (ViUR2 might have deleted them) 

276 skel = skeletonByKind("file")() 

277 if skel.fromDB(val["key"]): # This file-object exist, no need to recreate it 

278 return 

279 skel["key"] = val["key"] 

280 skel["name"] = val["name"] 

281 skel["mimetype"] = val["mimetype"] 

282 skel["dlkey"] = val["dlkey"] 

283 skel["size"] = val["size"] 

284 skel["width"] = val["width"] 

285 skel["height"] = val["height"] 

286 skel["weak"] = True 

287 skel["pending"] = False 

288 k = skel.toDB() 

289 

290 from viur.core.modules.file import importBlobFromViur2 

291 super().refresh(skel, boneName) 

292 if conf.get("viur.viur2import.blobsource"): 

293 # Just ensure the file get's imported as it may not have an file entry 

294 val = skel[boneName] 

295 if isinstance(val, list): 

296 for x in val: 

297 importBlobFromViur2(x["dest"]["dlkey"], x["dest"]["name"]) 

298 recreateFileEntryIfNeeded(x["dest"]) 

299 elif isinstance(val, dict): 

300 if not "dest" in val: 

301 return 

302 importBlobFromViur2(val["dest"]["dlkey"], val["dest"]["name"]) 

303 recreateFileEntryIfNeeded(val["dest"])
