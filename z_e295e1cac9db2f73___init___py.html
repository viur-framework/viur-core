98def __build_app(modules: ModuleType | object, renderers: ModuleType | object, default: str = None) -> Module: 

99 """ 

100 Creates the application-context for the current instance. 

101 

102 This function converts the classes found in the *modules*-module, 

103 and the given renders into the object found at ``conf.main_app``. 

104 

105 Every class found in *modules* becomes 

106 

107 - instanced 

108 - get the corresponding renderer attached 

109 - will be attached to ``conf.main_app`` 

110 

111 :param modules: Usually the module provided as *modules* directory within the application. 

112 :param renderers: Usually the module *viur.core.renders*, or a dictionary renderName => renderClass. 

113 :param default: Name of the renderer, which will form the root of the application. 

114 This will be the renderer, which wont get a prefix, usually html. 

115 (=> /user instead of /html/user) 

116 """ 

117 if not isinstance(renderers, dict): 

118 # build up the dict from viur.core.render 

119 renderers, mod = {}, renderers 

120 

121 from viur.core.render.abstract import AbstractRenderer 

122 

123 for render_name, render_mod in vars(mod).items(): 

124 if inspect.ismodule(render_mod): 

125 for render_clsname, render_cls in vars(render_mod).items(): 

126 # this is "kinda hackish..." because ViUR 3's current renderer concept is pure bulls*t... 

127 if render_clsname == "DefaultRender": 

128 continue 

129 

130 if ( 

131 # test for a renderer 

132 (inspect.isclass(render_cls) and issubclass(render_cls, AbstractRenderer)) 

133 # bullsh*t, this must be entirely reworked! 

134 or render_clsname == "_postProcessAppObj" 

135 ): 

136 renderers.setdefault(render_name, {}) 

137 renderers[render_name][render_clsname] = render_cls 

138 

139 # assign ViUR system modules 

140 from viur.core.modules.moduleconf import ModuleConf # noqa: E402 # import works only here because circular imports 

141 from viur.core.modules.script import Script # noqa: E402 # import works only here because circular imports 

142 from viur.core.modules.translation import Translation # noqa: E402 # import works only here because circular imports 

143 from viur.core.prototypes.instanced_module import InstancedModule # noqa: E402 # import works only here because circular imports 

144 

145 modules._tasks = TaskHandler 

146 modules._moduleconf = ModuleConf 

147 modules._translation = Translation 

148 modules.script = Script 

149 

150 # Resolver defines the URL mapping 

151 resolver = {} 

152 

153 # Index is mapping all module instances for global access 

154 index = (modules.index if hasattr(modules, "index") else Module)("index", "") 

155 index.register(resolver, renderers[default]["default"](parent=index)) 

156 

157 for module_name, module_cls in vars(modules).items(): # iterate over all modules 

158 if module_name == "index": 

159 continue # ignore index, as it has been processed before! 

160 

161 if module_name in renderers: 

162 raise NameError(f"Cannot name module {module_name!r}, as it is a reserved render's name") 

163 

164 if not ( # we define the cases we want to use and then negate them all 

165 (inspect.isclass(module_cls) and issubclass(module_cls, Module) # is a normal Module class 

166 and not issubclass(module_cls, InstancedModule)) # but not a "instantiable" Module 

167 or isinstance(module_cls, InstancedModule) # is an already instanced Module 

168 ): 

169 continue 

170 

171 # remember module_instance for default renderer. 

172 module_instance = default_module_instance = None 

173 

174 for render_name, render in renderers.items(): # look, if a particular renderer should be built 

175 # Only continue when module_cls is configured for this render 

176 # todo: VIUR4 this is for legacy reasons, can be done better! 

177 if not getattr(module_cls, render_name, False): 

178 continue 

179 

180 # Create a new module instance 

181 module_instance = module_cls( 

182 module_name, ("/" + render_name if render_name != default else "") + "/" + module_name 

183 ) 

184 

185 # Attach the module-specific or the default render 

186 if render_name == default: # default or render (sub)namespace? 

187 default_module_instance = module_instance 

188 target = resolver 

189 else: 

190 if getattr(index, render_name, True) is True: 

191 # Render is not build yet, or it is just the simple marker that a given render should be build 

192 setattr(index, render_name, Module(render_name, "/" + render_name)) 

193 

194 # Attach the module to the given renderer node 

195 setattr(getattr(index, render_name), module_name, module_instance) 

196 target = resolver.setdefault(render_name, {}) 

197 

198 module_instance.register(target, render.get(module_name, render["default"])(parent=module_instance)) 

199 

200 # Apply Renderers postProcess Filters 

201 if "_postProcessAppObj" in render: # todo: This is ugly! 

202 render["_postProcessAppObj"](target) 

203 

204 # Ugly solution, but there is no better way to do it in ViUR 3: 

205 # Allow that any module can be accessed by `conf.main_app.<modulename>`, 

206 # either with default render or the last created render. 

207 # This behavior does NOT influence the routing. 

208 if default_module_instance or module_instance: 

209 setattr(index, module_name, default_module_instance or module_instance) 

210 

211 # fixme: Below is also ugly... 

212 if default in renderers and hasattr(renderers[default]["default"], "renderEmail"): 

213 conf.emailRenderer = renderers[default]["default"]().renderEmail 

214 elif "html" in renderers: 

215 conf.emailRenderer = renderers["html"]["default"]().renderEmail 

216 

217 # This might be useful for debugging, please keep it for now. 

218 if conf.debug.trace: 

219 import pprint 

220 logging.debug(pprint.pformat(resolver)) 

221 

222 conf.main_resolver = resolver 

223 conf.main_app = index

226def setup(modules: ModuleType | object, render: ModuleType | object = None, default: str = "html"): 

227 """ 

228 Define whats going to be served by this instance. 

229 

230 :param modules: Usually the module provided as *modules* directory within the application. 

231 :param render: Usually the module *viur.core.renders*, or a dictionary renderName => renderClass. 

232 :param default: Name of the renderer, which will form the root of the application.\ 

233 This will be the renderer, which wont get a prefix, usually html. \ 

234 (=> /user instead of /html/user) 

235 """ 

236 from viur.core.bones.base import setSystemInitialized 

237 # noinspection PyUnresolvedReferences 

238 import skeletons # This import is not used here but _must_ remain to ensure that the

243 if not render: 

244 import viur.core.render 

245 render = viur.core.render 

246 

247 __build_app(modules, render, default) 

248 

249 # Send warning email in case trace is activated in a cloud environment 

250 if ((conf.debug.trace 

251 or conf.debug.trace_external_call_routing 

252 or conf.debug.trace_internal_call_routing) 

253 and (not conf.instance.is_dev_server or conf.debug.dev_server_cloud_logging)): 

254 from viur.core import email

256 email.sendEMailToAdmins( 

257 "Debug mode enabled", 

258 "ViUR just started a new Instance with call tracing enabled! This might log sensitive information!" 

259 ) 

260 except Exception as exc: # OverQuota, whatever 

261 logging.exception(exc) 

262 # Ensure that our Content Security Policy Header Cache gets build 

263 from viur.core import securityheaders 

264 securityheaders._rebuildCspHeaderCache() 

265 securityheaders._rebuildPermissionHeaderCache() 

266 setSystemInitialized() 

267 # Assert that all security related headers are in a sane state 

268 if conf.security.content_security_policy and conf.security.content_security_policy["_headerCache"]: 

269 for k in conf.security.content_security_policy["_headerCache"]: 

270 if not k.startswith("Content-Security-Policy"): 

271 raise AssertionError("Got unexpected header in " 

272 "conf.security.content_security_policy['_headerCache']") 

273 if conf.security.strict_transport_security: 

274 if not conf.security.strict_transport_security.startswith("max-age"): 

275 raise AssertionError("Got unexpected header in conf.security.strict_transport_security") 

276 crossDomainPolicies = {None, "none", "master-only", "by-content-type", "all"} 

277 if conf.security.x_permitted_cross_domain_policies not in crossDomainPolicies: 

278 raise AssertionError("conf.security.x_permitted_cross_domain_policies " 

279 f"must be one of {crossDomainPolicies!r}") 

280 if conf.security.x_frame_options is not None and isinstance(conf.security.x_frame_options, tuple): 

281 mode, uri = conf.security.x_frame_options 

282 assert mode in ["deny", "sameorigin", "allow-from"] 

283 if mode == "allow-from": 

284 assert uri is not None and (uri.lower().startswith("https://") or uri.lower().startswith("http://")) 

285 runStartupTasks() # Add a deferred call to run all queued startup tasks 

286 i18n.initializeTranslations() 

287 if conf.file_hmac_key is None: 

288 from viur.core import db 

289 key = db.Key("viur-conf", "viur-conf") 

290 if not (obj := db.Get(key)): # create a new "viur-conf"? 

291 logging.info("Creating new viur-conf") 

292 obj = db.Entity(key) 

293 

294 if "hmacKey" not in obj: # create a new hmacKey 

295 logging.info("Creating new hmacKey") 

296 obj["hmacKey"] = utils.string.random(length=20) 

297 db.Put(obj) 

298 

299 conf.file_hmac_key = bytes(obj["hmacKey"], "utf-8") 

300 

301 if conf.instance.is_dev_server: 

302 WIDTH = 80 # defines the standard width 

303 FILL = "#" # define sthe fill char (must be len(1)!) 

304 PYTHON_VERSION = (sys.version_info.major, sys.version_info.minor, sys.version_info.micro) 

305 

306 # define lines to show 

307 lines = ( 

308 " LOCAL DEVELOPMENT SERVER IS UP AND RUNNING ", # title line 

309 f"""project = \033[1;31m{conf.instance.project_id}\033[0m""", 

310 f"""python = \033[1;32m{".".join((str(i) for i in PYTHON_VERSION))}\033[0m""", 

311 f"""viur = \033[1;32m{".".join((str(i) for i in conf.version))}\033[0m""", 

312 "" # empty line 

313 ) 

314 

315 # first and last line are shown with a cool line made of FILL 

316 first_last = (0, len(lines) - 1) 

317 

318 # dump to console

321 f"""\033[0m{FILL}{line:{ 

322 FILL if i in first_last else " "}^{(WIDTH - 2) + (11 if i not in first_last else 0) 

323 }}{FILL}""" 

324 )
